83 {
84 if (m.size() == 0)
85 {
86 numParams = params.size();
87 for (std::shared_ptr<Eigen::Tensor<DT, Dim>>& param : params)
88 {
89 std::shared_ptr<Eigen::Tensor<DT, Dim>> m0 = std::make_shared<Eigen::Tensor<DT, Dim>>(param->getData()->dimensions());
90 m0->setZero();
91 std::shared_ptr<Eigen::Tensor<DT, Dim>> v0 = std::make_shared<Eigen::Tensor<DT, Dim>>(param->getData()->dimensions());
92 v0->setZero();
93
94 m.push_back(m0);
95 v.push_back(v0);
96 }
97 }
98 else if (m.size() != params.size())
99 {
100 throw std::runtime_error("Number of parameters changed between updates!");
101 }
102
103 t++;
104
105 for (int32_t i = 0; i < numParams; i++)
106 {
107 *m[i] = (beta1 * *m[i]) + ((1 - beta1) * *params[i]->getGrad());
108 *v[i] = (beta2 * *v[i]) + ((1 - beta2) * (params[i]->getGrad()->pow(2)));
109
110 Eigen::Tensor<DT, Dim> mHat = *m[i] / (1 - std::pow(beta1, t));
111 Eigen::Tensor<DT, Dim> vHat = *v[i] / (1 - std::pow(beta2, t));
112
113 *params[i]->getData() = *params[i]->getData() - (learningRate * (mHat / (vHat.sqrt() + epsilon)));
114 params[i]->zeroGrad();
115 }
116 }

120 m.clear();
121 v.clear();
122 t = 0;
123 }
