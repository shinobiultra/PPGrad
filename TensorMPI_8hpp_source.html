55 template <int Dim, typename DT>
56 std::vector<std::shared_ptr<TensorBase<2, double>>> tensorScatter(
57 std::vector<Eigen::Tensor<DT, Dim>> &inputs,
58 const int worldSize,
59 const int worldRank)
60 {
61 std::vector<std::shared_ptr<TensorBase<Dim, DT>>> outputs;
62
63 // Flatten the data on the root process
64 std::vector<DT> flat_data;
65 if (worldRank == 0)
66 {
67 for (const auto &tensor : inputs)
68 {
69 flat_data.insert(flat_data.end(), tensor.data(), tensor.data() + tensor.size());
70 }
71 }
72
73 // Broadcast the number of elements to send to each process
74 int sendCount;
75 if (worldRank == 0)
76 {
77 // Calculate the number of elements to send to each process
78 sendCount = flat_data.size() / worldSize;
79 }
80 MPI_Bcast(&sendCount, 1, MPI_INT, 0, MPI_COMM_WORLD);
81
82 // Issue a graceful warning if the data is not evenly divisible amongst the processes
83 if (flat_data.size() % worldSize != 0)
84 {
85 std::cerr << "Warning: Data is not evenly divisible amongst processes." << std::endl;
86
87#ifdef PPGRAD_DEBUG
88 std::cerr << "Data size: " << flat_data.size() << std::endl;
89 std::cerr << "World size: " << worldSize << std::endl;
90 std::cerr << "Send count: " << sendCount << std::endl;
91 std::cerr << "Input Tensors: " << inputs.size() << std::endl;
92#endif
93 }
94
95 // Create a receive buffer
96 std::vector<double> recv_data(sendCount);
97
98 // Scatter the data
99 MPI_Scatter(flat_data.data(), sendCount, MPI_DOUBLE, recv_data.data(), sendCount, MPI_DOUBLE, 0, MPI_COMM_WORLD);
100
101 // Distribute the dimensions of the tensors (assuming they are all the same size)
102 std::vector<int> tensorDims;
103 int tensorDimLength;
104
105 if (worldRank == 0)
106 {
107 auto &dimensions = inputs[0].dimensions();
108 for (size_t i = 0; i < dimensions.size(); i++)
109 {
110 tensorDims.push_back(dimensions[i]);
111 }
112 tensorDimLength = dimensions.size();
113 }
114
115 // Distribute the lengths of the dimensions
116 MPI_Bcast(&tensorDimLength, 1, MPI_INT, 0, MPI_COMM_WORLD);
117
118 // Distribute the dimensions
119 if (worldRank != 0)
120 {
121 tensorDims.resize(tensorDimLength);
122 }
123 MPI_Bcast(tensorDims.data(), tensorDimLength, MPI_INT, 0, MPI_COMM_WORLD);
124
125 int tensorSize = tensorDims[0];
126 for (size_t i = 1; i < tensorDims.size(); i++)
127 tensorSize *= tensorDims[i];
128
129 // Unflatten the data
130 for (size_t i = 0; i < recv_data.size(); i += tensorSize)
131 {
132 Eigen::TensorMap<Eigen::Tensor<DT, Dim>> tensorMap = createTensorMap<Dim, DT>(recv_data.data() + i, tensorDims);
133 Eigen::Tensor<DT, Dim> tensor = tensorMap;
134 outputs.push_back(std::make_shared<Tensor<Dim, DT>>(std::make_shared<Eigen::Tensor<DT, Dim>>(tensor)));
135 }
136
137 return outputs;
138 }

148 template <int Dim, typename DT>
149 std::shared_ptr<PPNN::Model<Dim, DT>> modelBroadcast(
150 std::shared_ptr<PPNN::Model<Dim, DT>> model,
151 const int worldSize,
152 const int worldRank)
153 {
154 // Make sure all nodes have the same number of parameters
155 int paramCount = model->getParams().size();
156 MPI_Allreduce(MPI_IN_PLACE, &paramCount, 1, MPI_INT, MPI_SUM, MPI_COMM_WORLD);
157 paramCount /= worldSize;
158
159 if (paramCount != model->getParams().size())
160 {
161 std::cerr << "Error: All nodes must have the same number of parameters in the model before broadcasting." << std::endl;
162 MPI_Abort(MPI_COMM_WORLD, 1);
163 }
164
165 // Broadcast the parameters
166 for (size_t i = 0; i < paramCount; i++)
167 {
168 // Get the data from the model
169 std::vector<int> tensorDims;
170 for (int dim : model->getParams()[i]->getData()->dimensions())
171 tensorDims.push_back(dim);
172
173 Eigen::TensorMap<Eigen::Tensor<DT, Dim>> tensorMap = createTensorMap<Dim, DT>(model->getParams()[i]->getData()->data(), tensorDims);
174 Eigen::Tensor<DT, Dim> tensor = tensorMap;
175
176 // Broadcast the data
177 MPI_Bcast(tensor.data(), tensor.size(), MPI_DOUBLE, 0, MPI_COMM_WORLD);
178
179 // Set the data in the model
180 *model->getParams()[i]->getData() = tensor;
181 }
182
183 return model;
184 }
